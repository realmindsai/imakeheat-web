# SP-404 Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `isolator`, `equalizer`, `filterDrive`, `compressor`, and `loFi` as real pedalboard effects using the existing slot registry and render path.

**Architecture:** Keep every Phase 1 effect inside the current `EffectDefinition.build()` contract. Use native-node composite chains for `isolator`, `equalizer`, `filterDrive`, and `compressor`, and build `loFi` by composing the already-loaded `srhold` and `bitcrusher` worklets with native filters and a dry/wet wrapper. Defaults must be bypass-safe so neutral slots are omitted from the chain by `isNeutral()`.

**Tech Stack:** TypeScript, React 18, Zustand, Web Audio API, AudioWorklet, Vite, Vitest, Playwright.

**Spec:** `docs/superpowers/specs/2026-05-01_sp404-phase-1-design.md`
**Baseline:** current `main` with `303 VinylSim`, `404 VinylSim`, and `Cassette Sim` already shipped.

## Defaults And Mappings

Lock these defaults before implementation so `isNeutral()` and tests agree:

- `isolator`: `{ low: 0, mid: 0, high: 0 }`
- `equalizer`: `{ lowGain: 0, midGain: 0, highGain: 0, lowFreq: 80, midFreq: 1000, highFreq: 8000 }`
- `filterDrive`: `{ cutoffHz: 16000, resonance: 0, drive: 0, filterType: 'lowpass', lowFreq: 200, lowGain: 0 }`
- `compressor`: `{ sustain: 0, attack: 50, ratio: 0, level: 100 }`
- `loFi`: `{ preFilt: 1, lofiType: 1, tone: 0, cutoffHz: 8000, balance: 0, level: 100 }`

Map UI ranges like this:

- `isolator` gains: `-60..12`
- `equalizer` gains: `-15..15`
- `filterDrive.cutoffHz`: `20..16000`
- `filterDrive.resonance`: `0..100` mapped internally to `Q = 0.0001..16`
- `filterDrive.drive`: `0..100`
- `filterDrive.lowFreq`: `20..16000`
- `filterDrive.lowGain`: `-24..24`
- `compressor.sustain`: `0..100`
- `compressor.attack`: `0..100`
- `compressor.ratio`: `0..100`
- `compressor.level`: `0..100`
- `loFi.preFilt`: `1..6`
- `loFi.lofiType`: `1..9`
- `loFi.tone`: `-100..100`
- `loFi.cutoffHz`: `200..8000`
- `loFi.balance`: `0..100` where `0=dry` and `100=wet`
- `loFi.level`: `0..100`

```ts
export interface ThreeBandEqNodes {
  input: BiquadFilterNode
  low: BiquadFilterNode
  mid: BiquadFilterNode
  high: BiquadFilterNode
  output: BiquadFilterNode
}

export function createThreeBandEq(
  ctx: BaseAudioContext,
  freqs: { low: number; mid: number; high: number },
): ThreeBandEqNodes {
  const low = ctx.createBiquadFilter()
  low.type = 'lowshelf'
  low.frequency.value = freqs.low

  const mid = ctx.createBiquadFilter()
  mid.type = 'peaking'
  mid.frequency.value = freqs.mid
  mid.Q.value = 0.707

  const high = ctx.createBiquadFilter()
  high.type = 'highshelf'
  high.frequency.value = freqs.high

  low.connect(mid)
  mid.connect(high)

  return { input: low, low, mid, high, output: high }
}

export function setThreeBandEqGains(
  nodes: ThreeBandEqNodes,
  gains: { low: number; mid: number; high: number },
  currentTime: number,
) {
  nodes.low.gain.setTargetAtTime(gains.low, currentTime, 0.01)
  nodes.mid.gain.setTargetAtTime(gains.mid, currentTime, 0.01)
  nodes.high.gain.setTargetAtTime(gains.high, currentTime, 0.01)
}
```

```ts
import { IsolatorPanel } from './panel'

type P = { low: number; mid: number; high: number }

const def: EffectDefinition<'isolator'> = {

  defaultParams: { low: 0, mid: 0, high: 0 },

      output: nodes.output,

      dispose() {
        nodes.low.disconnect()
```

```ts
export function makeSoftClipCurve(amount: number): Float32Array {
  const curve = new Float32Array(2048)
  const k = Math.max(0, amount)
  for (let i = 0; i < curve.length; i++) {
    const x = (i / (curve.length - 1)) * 2 - 1
    curve[i] = Math.tanh((1 + k * 3) * x) / Math.tanh(1 + k * 3)
  }
  return curve
}

export function outputTrimFromDrive(amount: number): number {
  return 1 / (1 + amount * 0.35)
}
```

```ts
import { register } from '../_internal'
import type { EffectDefinition, EffectNode } from '../types'
import { makeSoftClipCurve, outputTrimFromDrive } from '../_shared/waveshaper'
import { FilterDrivePanel } from './panel'

type P = {
  cutoffHz: number
  resonance: number
  drive: number
  filterType: 'lowpass' | 'highpass'
  lowFreq: number
  lowGain: number
}

const def: EffectDefinition<'filterDrive'> = {
  kind: 'filterDrive',
  displayName: 'Filter+Drive',
  defaultParams: { cutoffHz: 16000, resonance: 0, drive: 0, filterType: 'lowpass', lowFreq: 200, lowGain: 0 },
  isNeutral: (p) =>
    p.cutoffHz === 16000
    && p.resonance === 0
    && p.drive === 0
    && p.filterType === 'lowpass'
    && p.lowGain === 0
    && p.lowFreq === 200,
  build(ctx, params): EffectNode<P> {
    const filter = ctx.createBiquadFilter()
```

```ts
import { register } from '../_internal'
import type { EffectDefinition, EffectNode } from '../types'
import { CompressorPanel } from './panel'

type P = { sustain: number; attack: number; ratio: number; level: number }

function ratioValue(ratio: number) {
  return 1.5 + (ratio / 100) * 10.5
}

function attackValue(attack: number) {
  return 0.003 + (attack / 100) * 0.117
}

function thresholdValue(sustain: number) {
  return -2 - (sustain / 100) * 28
}

function makeupGain(level: number, sustain: number) {
  const base = level / 100
  const makeup = 1 + (sustain / 100) * 0.6
  return base * makeup
}

const def: EffectDefinition<'compressor'> = {
  kind: 'compressor',
  displayName: 'Compressor',
  defaultParams: { sustain: 0, attack: 50, ratio: 0, level: 100 },
  isNeutral: (p) => p.sustain === 0 && p.attack === 50 && p.ratio === 0 && p.level === 100,
  build(ctx, params): EffectNode<P> {
    const comp = ctx.createDynamicsCompressor()
    const out = ctx.createGain()
    comp.connect(out)
    const apply = (p: P) => {
      comp.threshold.setTargetAtTime(thresholdValue(p.sustain), ctx.currentTime, 0.01)
      comp.attack.setTargetAtTime(attackValue(p.attack), ctx.currentTime, 0.01)
      comp.ratio.setTargetAtTime(ratioValue(p.ratio), ctx.currentTime, 0.01)
      comp.knee.setTargetAtTime(18, ctx.currentTime, 0.01)
      comp.release.setTargetAtTime(0.12, ctx.currentTime, 0.01)
      out.gain.setTargetAtTime(makeupGain(p.level, p.sustain), ctx.currentTime, 0.01)
    }
    apply(params)
    return { input: comp, output: out, apply, dispose() { comp.disconnect(); out.disconnect() } }
  },
```

```ts
import type { EffectDefinition, EffectNode } from '../types'
import { setDryWet } from '../_shared/dryWet'
import { LoFiPanel } from './panel'

type P = { preFilt: number; lofiType: number; tone: number; cutoffHz: number; balance: number; level: number }

function crusherBits(lofiType: number) {
  return ([16, 12, 12, 8, 8, 6, 4, 4, 2] as const)[lofiType - 1]
}

function heldRate(ctx: BaseAudioContext, lofiType: number) {
  return [44100, 32000, 24000, 18000, 12000, 8000, 6000, 4000, 3000][lofiType - 1] ?? ctx.sampleRate
}

function preFreq(preFilt: number) {
  return [16000, 9000, 6000, 3200, 1800, 900][preFilt - 1] ?? 16000
}

const def: EffectDefinition<'loFi'> = {
  kind: 'loFi',
  displayName: 'Lo-fi',
  defaultParams: { preFilt: 1, lofiType: 1, tone: 0, cutoffHz: 8000, balance: 0, level: 100 },
  isNeutral: (p) => p.preFilt === 1 && p.lofiType === 1 && p.tone === 0 && p.cutoffHz === 8000 && p.balance === 0 && p.level === 100,
  build(ctx, params): EffectNode<P> {
    const input = ctx.createGain()
    const output = ctx.createGain()
    const dry = ctx.createGain()
    const wetOut = ctx.createGain()
    const pre = ctx.createBiquadFilter()
    const tone = ctx.createBiquadFilter()
    const cutoff = ctx.createBiquadFilter()
    const level = ctx.createGain()
    const srhold = new AudioWorkletNode(ctx, 'srhold')
    const crusher = new AudioWorkletNode(ctx, 'bitcrusher')

    input.connect(dry)
    input.connect(pre)
    pre.connect(srhold)
    srhold.connect(crusher)
    crusher.connect(tone)
    tone.connect(cutoff)
    cutoff.connect(level)
    level.connect(wetOut)
    dry.connect(output)
    wetOut.connect(output)

    tone.type = 'highshelf'
    tone.frequency.value = 2200
    cutoff.type = 'lowpass'

    const apply = (p: P) => {
      pre.type = 'lowpass'
      pre.frequency.setTargetAtTime(preFreq(p.preFilt), ctx.currentTime, 0.01)
      tone.gain.setTargetAtTime((p.tone / 100) * 12, ctx.currentTime, 0.01)
      cutoff.frequency.setTargetAtTime(p.cutoffHz, ctx.currentTime, 0.01)
      level.gain.setTargetAtTime(p.level / 100, ctx.currentTime, 0.01)
      srhold.port.postMessage({ sampleRateHz: heldRate(ctx, p.lofiType) })
      crusher.port.postMessage({ bitDepth: crusherBits(p.lofiType) })
```
